     static string Pattern(Color c,string s) {      
      string[] sa=s.Split(',');
      int w=1;
      foreach(string l in sa) if(l.Length>w) w=l.Length;
      Bitmap b=new Bitmap(w,sa.Length,PixelFormat.Format32bppArgb);
      using(Graphics g=Graphics.FromImage(b)) {
        g.Clear(Color.Transparent);
      }
      for(int y=0,x=0;y<sa.Length;y++,x=0) {
        foreach(char ch in sa[y]) {
          if(ch=='x') b.SetPixel(x,y,c);
          x++;
        }
      }
      MemoryStream ms=new MemoryStream();
      b.Save(ms,ImageFormat.Png);
      byte[] img=ms.GetBuffer();
      return Convert.ToBase64String(img,0,(int)ms.Length);
    }
